impl SparseVec {

    #[inline]
    pub fn overlap(&self, other: &SparseVec) -> u32 {
        let (mut i, mut j, mut c) = (0usize, 0usize, 0u32);
        let (a, b) = (&self.indices, &other.indices);

    /// One scale's Hopfield retrieval. Returns ranked candidate ids.
    /// This IS the gradient step on E = -(1/beta) lse(beta, overlaps).
    fn retrieve_scale(&self, scale: usize, query: &SparseVec, top: usize)
        -> Vec<(u32, f32)>
    {
        // STEP A: overlap scan via inverted index вҖ” only touch items
        //         that share >=1 active index with the query. O(k * fanout).
        let mut acc: std::collections::HashMap<u32, u32> = Default::default();
        for &d in &query.indices {
            for &id in &self.inv[scale][d as usize] {
                *acc.entry(id).or_insert(0) += 1; // accumulate overlap
            }
        }
        if acc.is_empty() { return vec![]; }

        // STEP B: softmax attention over overlaps (numerically stable).
        let max_o = *acc.values().max().unwrap() as f32;
        let mut scored: Vec<(u32, f32)> = acc.into_iter()
            .map(|(id, o)| (id, (self.beta * (o as f32 - max_o)).exp()))
            .collect();
        let z: f32 = scored.iter().map(|(_, w)| *w).sum();
        for (_, w) in &mut scored { *w /= z; }

        // STEP C: top-k projection (rank by attention weight).
        scored.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
        scored.truncate(top);
        scored
        // Convergence: re-running with the top-1 candidate as the new query
        // is the second iteration. For k-sparse it converges in 1-3 steps
        // (proof in Section 4). We expose `complete()` to iterate.
    }

    /// Full multi-scale completion: cascade coarse -> fine.
    /// Returns the FINEST scale that recovered, plus its id.
    /// This is where graceful degradation lives.
    pub fn complete(&self, q_fine: &SparseVec, q_med: &SparseVec,
                    q_coarse: &SparseVec) -> Recovered
    {

        // return leaf identity.
        let f = self.retrieve_scale(0, q_fine, 10);

            return Recovered::Leaf(f[0].0);

        let m = self.retrieve_scale(1, q_med, 10);

            return Recovered::SubCluster(m[0].0); // less info, still useful

        // Medium failed. Coarse is the last line вҖ” return category only.
        let c = self.retrieve_scale(2, q_coarse, 10);
        if decisive(&c) {
            return Recovered::Category(c[0].0);
        }
        Recovered::Nothing
    }

}
