pub fn certify_basin(mem: &HolographicMemory, n: usize, beta: f32) -> Certification {

    // ACTUAL stored index distribution (not the idealized Poisson lambda).

/// then ask the attractor to complete. Pure test of recovery-from-fragment.

pub struct LsmMemtable {
    pending_updates: Vec<(u32, SparseVec, SparseVec)>, // (id, old_dims, new_dims)
    dim_fanout: Vec<u32>, // Current fanout of all D dimensions

impl LsmMemtable {
    /// Evaluates the interference gradient before flushing to the SSTable.
    /// Returns `true` if the update is safe, `false` if it must be rejected
    /// or routed to the Polysemy Spawner (Debate D's Jaccard < 0.25 rule).
    pub fn check_interference_guard(&mut self, id: u32, old: &SparseVec, new: &SparseVec) -> bool {
        let added_dims: Vec<u32> = new.indices.iter()
            .filter(|d| !old.indices.contains(d))
            .cloned()
            .collect();

        // Calculate the worst-case Delta-Tau injected into the index.
        // Adding pattern `id` to dimension `d` increases the overlap between
        // `id` and ANY other pattern `k` that also uses `d` by exactly 1.
        // The danger is when `id` shares multiple NEW dimensions with a 
        // single existing pattern `k`, spiking their pairwise overlap.
        
        // We approximate the max pairwise interference by checking the 
        // existing fanout of the new dimensions. High fanout = high probability
        // of colliding with an existing pattern's active set.
        let mut max_delta_tau = 0.0;
        for &d in &added_dims {
            let current_fanout = self.dim_fanout[d as usize];
            
            // The probability that a random existing pattern uses `d` is 
            // fanout / N. The expected number of collisions in k dims is 
            // k * (fanout / N). We use the Poisson extreme-value bound 
            // to estimate the max overlap contribution of this dimension.
            let lambda_d = current_fanout as f32 / N as f32;
            
            // If this dimension is too common (low IDF), it acts as noise.
            // This is the mathematical core of Debate E's IDF clipping.
            if lambda_d > 0.05 { 
                return false; // REJECT: Dimension is too dense, kills basins.
            }
            
            max_delta_tau += lambda_d; 
        }

        // If the cumulative interference pushes the global tau estimate 
        // over the certified cliff, block the write.

            return false; // REJECT: Write-path poisoning detected.

        for &d in &added_dims { self.dim_fanout[d as usize] += 1; }
        for &d in &old.indices { 
            if !new.indices.contains(&d) { self.dim_fanout[d as usize] -= 1; } 
        }
        true
    }
}
